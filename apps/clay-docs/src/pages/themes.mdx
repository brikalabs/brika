    <span class="spacer hairline"></span>
  </div>
  <h2 id="apply" class="mb-3 scroll-mt-24 font-display text-3xl text-clay-strong leading-[1.1]">
    Theme the whole app
  </h2>
  <p class="mb-5 max-w-2xl text-clay-default text-sm leading-relaxed">
    `applyTheme(theme, mode)` writes every colour onto `document.documentElement` as a CSS
    custom property. Every Clay component on the page reads the new values immediately — no
    re-render, no React state.
  </p>
  <CodeSource
    code={`import { nord, applyTheme } from "@brika/clay/themes";

applyTheme(nord, "dark");`}
    lang="tsx"
    filename="globally"
  />
</section>

<section class="mb-16">

    <span class="spacer hairline"></span>
  </div>
  <h2 id="scope" class="mb-3 scroll-mt-24 font-display text-3xl text-clay-strong leading-[1.1]">
    Or scope it to a subtree
  </h2>
  <p class="mb-5 max-w-2xl text-clay-default text-sm leading-relaxed">
    `themeToCssVars(theme, mode)` returns a `CSSProperties` object you can spread onto any
    element's style prop. The theme applies only to that subtree — useful for side-by-side
    comparisons like the gallery above.
  </p>
  <CodeSource
    code={`import { ocean, themeToCssVars } from "@brika/clay/themes";

<div style={themeToCssVars(ocean, "light")}>
  <Button>Ocean button</Button>
</div>`}
    lang="tsx"
    filename="scoped"
  />
</section>

  <h2 id="shape" class="mb-3 scroll-mt-24 font-display text-3xl text-clay-strong leading-[1.1]">
    ThemeConfig
  </h2>
  <p class="mb-5 max-w-2xl text-clay-default text-sm leading-relaxed">
    Every preset is a typed object. The registry exports BUILT_IN_THEMES (ordered),
    BUILT_IN_THEMES_BY_ID (lookup map), and each theme by name — nord, ocean, forest, sunset,
    lavender, ruby, solarized, candy, dracula, mono, plus default_ as the reserved-word rename.
  </p>
  <CodeSource
    code={`interface ThemeConfig {
  id: string;
  name: string;
  description: string;
  accentSwatches: readonly string[];
  colors: {
    light: Readonly<Record<string, string>>;
    dark: Readonly<Record<string, string>>;
  };
}`}
    lang="ts"
    filename="types.ts"
  />
</section>
